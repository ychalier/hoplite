 76        """Compute the infinite norm of the vector.

 86    def gradient(self, other):
 87        """Compute the direction between two positions.
 88
 89        Parameters
 90        ----------
 91        other : HexagonalCoordinates

 93

 99        """

125    def rotate(self, steps):
126        """Rotate an hexagonal vector.
127
128        Parameters
129        ----------
130        steps : int
131            Number of rotation steps. One step corresponds to a angle of pi/3,
132            i.e. one sixth of a circle, in trigonometric direction
133            (counterclockwise). A negative number of steps will rotate
134            clockwise.
135
136        Returns
137        -------
138        HexagonalCoordinates
139            Rotated vector.
140
141        """
142        clockwise = steps < 0
143        result = self.copy()
144        for _ in range(abs(steps)):
145            if clockwise:
146                result = HexagonalCoordinates(-result.z, -result.x)
147            else:
148                result = HexagonalCoordinates(-result.y, -result.z)
149        return result

153    """Iterates over the coordinates of the map.
154
155    Returns
156    -------
157    Iterator[HexagonalCoordinates]
158        Iterator over the 79 tiles of the map.
159
160    """
161    # pylint: disable=C0103
162    for x, height in zip(range(-4, 5), [7, 8, 9, 10, 11, 10, 9, 8, 7]):
163        if x >= 0:
164            start = -5
165        else:
166            start = - 5 - x
167        for y in range(start, start + height):

182def hexagonal_neighbors(pos):
183    """Return the set of surrounding positions in an hexagonal grid.
184
185    Parameters
186    ----------
187    pos : HexagonalCoordinates
188        Center position.
189
190    Returns
191    -------
192    set[HexagonalCoordinates]
193        Set of positions surrounding the center.
194
195    """
196    return set(pos + direction for direction in HEXAGONAL_DIRECTIONS)\
197        .intersection(SURFACE_COORDINATES)

200def hexagonal_circle(center, radius):
201    """Compute an hexagonal circle given a center and a radius.
202
203    Parameters
204    ----------
205    center : HexagonalCoordinates
206        Center of the hexagonal circle.
207    radius : int
208        Radius of the hexagonal circle (using the infinite norm
209        of `HexagonalCoordinates.norm`).
210
211    Returns
212    -------
213    set[HexagonalCoordinates]
214        Set of positions within the hexagonal circle of given center and radius.
215
216    """
217    return {
218        pos for pos in SURFACE_COORDINATES
219        if (pos - center).norm() <= radius
220    }

223def hexagonal_line(start, direction):
224    """Compute an hexagonal line given a starting position and a direction.
225
226    Parameters
227    ----------
228    start : HexagonalCoordinates
229        Beginning of the line.
230    direction : HexagonalCoordinates
231        Direction (vector of norm 1) of the line.
232
233    Returns
234    -------
235    list[HexagonalCoordinates]
236        Tiles within the hexagonal line, from `start` and going.
237
238    """
239    result = [start]
240    while True:
241        current = result[-1] + direction
242        if current not in SURFACE_COORDINATES:
243            break
244        result.append(current)
245    return result
